# 시퀀스 다이어그램
$$uml
(웹브라우저 -> 웹서버 : 요청)
(웹브라우저 <- 웹서버 : 응답) as (UC2)
$$

# 자바 코드
```java
class Solution {
    public int solution(int hp) {
        int generalAntsCount = 0;
        int soldierAntsCount = 0;
        int workerAntsCount = 0;

        generalAntsCount = hp / 5;
        hp %= 5;

        soldierAntsCount = hp / 3;
        hp %= 3;

        workerAntsCount = hp;

        return generalAntsCount + soldierAntsCount + workerAntsCount;
    }

} 
```
